\problem Find the following derivatives (you do not have to provide arguments for this problem)
    \begin{enumerate}[(a)]
        \item $\frac{d}{dx} x^2$
        \item $\frac{d}{dx}\sin x$
        \item $\frac{d}{dx}\cos x$
        \item $\frac{d}{dx}e^x$
        \item $\frac{d}{dx}\ln x$
    \end{enumerate} \vspace{6mm}

\problem Find the following derivatives
    \begin{enumerate}[(a)]
        \item $\frac{d}{dx}e^2$
        \item $\frac{d}{dx}\frac{1}{\sqrt[5]{x^2}}$
        \item $\frac{d}{dx}(x^2 \sin x)$
        \item $\frac{d}{dx}\frac{x+2}{x^2+1}$
        \item $\frac{d}{dx}\csc x$
    \end{enumerate} \vspace{6mm}
    
\problem Find the following derivatives
    \begin{enumerate}[(a)]
        \item $\frac{d}{dx}\sqrt{\arctan x}$
        \item $\frac{d}{dx} e^{\cos x}$
        \item $\frac{d}{dx}\ln(\ln x)$
        \item $\frac{d}{dx}\cos(x \tan x)$
        \item $\frac{d}{dx}\sin(e^{x^2+1})$
    \end{enumerate} \vspace{6mm}

\problem Let $f(x) = 2x^3 - 3x^2 - 12x + 5$.
	\begin{enumerate}[(a)]
    	\item Find all the critical points for $f(x)$.
    	\item Use the \textit{first derivative test} to show if $f(x)$ attains local maximum, minimum or neither at each of its critical points.
    	\item Use the \textit{second derivative test} to show if $f(x)$ attains local maximum, minimum or neither at each of its critical points.
    	\item Is global maximum or minimum attained at each of the critical points? Why?
	\end{enumerate}
