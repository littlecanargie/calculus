\bigskip
\noindent \textbf{Reminder:} Remember the $+C$!\\
\noindent \textbf{Note:} Your answer need not be exactly the hinted answer, as long as it is a legit antiderivative.
\bigskip

\problem Find the following antiderivative:
\[\int \frac{1}{x^2\sqrt{9-16x^2}}~dx\]
(Hint: You may use the identity $\int \csc^2 x~dx = - \cot x + C$)
\vspace{20mm}

\problem Find the following antiderivative:
\[\int \frac{1}{x^2\sqrt{9+16x^2}}~dx\]
(Hint: If you used trigonometric substitution, try transforming all functions into sines and cosines)
\vspace{20mm}

\problem Find the following antiderivative:
\[\int \frac{1}{x^2\sqrt{16x^2-9}}~dx\]
(Note: Assume $x > 3/4$ if you want to keep track of the absolute values)
\vspace{20mm}


\problem Find the following antiderivative:
\[\int \frac{1}{\sqrt{x^2-6x+13}}~dx\]
(Hint: Complete the square.  You may use the identity $\int \sec x~dx = \ln |\sec x + \tan x| + C$)
