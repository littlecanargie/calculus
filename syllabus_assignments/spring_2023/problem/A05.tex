\bigskip
\noindent \textbf{Reminder:} Remember the $+C$!\\
\noindent \textbf{Note:} Your answer need not be exactly the hinted answer, as long as it is a legit antiderivative.
\bigskip

\problem Find the following antiderivative:
\[\int (1-x^2)^{3/2}~dx\]
(Hint: One answer is $\frac{3}{8}\arcsin x + \frac{1}{8}x\sqrt{1-x^2}(5-2x^2) + C$)
\vspace{20mm}

\problem Find the following antiderivative:
\[\int \frac{\sqrt{1-x^2}}{x}~dx\]
(Hint: You may use $\int \csc x~dx= -\ln|\cot x + \csc x| + C$)
\vspace{20mm}

\problem Find the following antiderivative:
\[\int \frac{1}{\sqrt{4-9(2x+3)^2}}~dx\]
(Hint: What multiple of $\sin^2\theta$ should $9(2x+3)^2$ be to eliminate the square root?)
\vspace{20mm}


\problem Find the following antiderivative:
\[\int \sqrt{x-x^2}~dx\]
(Hint 1: Rewrite the expression in the square root so that it looks like $a^2-b^2(x-c)^2$)\\
(Hint 2: One answer is $\frac{1}{8}\arcsin(2x-1) - \frac{1}{4}(2x-1)\sqrt{x-x^2} + C$)
